QUICK START
═══════════

Try the new features:

# Run beginner example
python examples/01_basic_melodies.py

# Create a video
python examples/03_video_visualizations.py

# Test everything
python examples/quick_viz_test.py

# Use in your code
from algorythm.synth import Synth
from algorythm.export import Exporter
from algorythm.visualization import CircularVisualizer

synth = Synth(waveform='saw')
signal = synth.generate_note(frequency=440, duration=2.0)

exporter = Exporter()
visualizer = CircularVisualizer(sample_rate=44100)

# Export as MP4 video!
exporter.export(
    signal,
    'my_video.mp4',
    sample_rate=44100,
    visualizer=visualizer,
    video_width=1920,
    video_height=1080,
    video_fps=30
)
